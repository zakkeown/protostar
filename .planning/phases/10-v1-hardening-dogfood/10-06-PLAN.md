    2. Create `.github/workflows/changeset-required.yml`:
       ```yaml
       name: changeset-required
       on:
         pull_request:
           types: [opened, synchronize, reopened, labeled, unlabeled]
       jobs:
         check:
           runs-on: ubuntu-latest
           if: ${{ !contains(github.event.pull_request.labels.*.name, 'skip changeset') }}
           steps:
             - uses: actions/checkout@v4
               with:
                 fetch-depth: 0
             - name: detect packages/*/src changes
               id: detect
               run: |
                 git fetch origin ${{ github.base_ref }} --depth=1
                 if git diff --name-only origin/${{ github.base_ref }}...HEAD | grep -qE '^packages/[^/]+/src/'; then
                   echo "needs=1" &gt;&gt; "$GITHUB_OUTPUT"
                 else
                   echo "needs=0" &gt;&gt; "$GITHUB_OUTPUT"
                 fi
             - name: require changeset when src changes
               if: steps.detect.outputs.needs == '1'
               run: |
                 git diff --name-only origin/${{ github.base_ref }}...HEAD | grep -qE '^\.changeset/[^/]+\.md$' || {
                   echo "PR touches packages/*/src/** but ships no changeset (.changeset/*.md). Add one with \`pnpm changeset\` or label this PR \`skip changeset\`." &gt;&amp;2
                   exit 1
                 }
       ```
